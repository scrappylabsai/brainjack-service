Agent-Ready Install 
# Prerequisites — check these first
python3 --version   # need 3.10+
git --version
xcode-select -p     # macOS only

# Install BrainJack (macOS / Linux)
git clone https://github.com/scrappylabsai/brainjack-service.git
cd brainjack-service
./install.sh

# Install BrainJack (Windows PowerShell)
# git clone https://github.com/scrappylabsai/brainjack-service.git
# cd brainjack-service
# .\install.ps1

# Post-install (macOS): grant Accessibility permission
# System Settings → Privacy & Security → Accessibility → add "BrainJack"

# Verify
./docs/verify-macos.sh

# Your auth token is in .env (BRAINJACK_TOKEN=...)
# Service runs on :9898 — scan the QR code with the iOS app
